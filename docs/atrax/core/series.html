 57        if all(isinstance(x, int) for x in self.data):
 58            return "int"

 60            return "float"
 61        elif all(isinstance(x, bool) for x in self.data):

 63        elif all(isinstance(x, datetime) for x in self.data):

233    def to_datetime(self, format='%Y-%m-%d', errors='raise'):
234        """
235        Convert the Series to datetime objects.
236        
237        Parameters:
238        format (str): The format of the date strings. Defaults to '%m/%d/%Y'.
239        errors (str): 'raise' to throw errors, 'coerce' to return None on failure
240        
241        Returns:
242        Series: A new Series with datetime objects.
243        """
244        converted = []
245
246        for val in self.data:
247            if isinstance(val, datetime):
248                converted.append(val)
249            elif isinstance(val, str):
250                try:
251                    converted.append(datetime.strptime(val, format))
252                except Exception as e:
253                    if errors == 'coerce':
254                        converted.append(None)
255                    else:
256                        raise ValueError(f"Failed to parse '{val}' as datetime: {e}")
257            else:
258                if errors == 'coerce':
259                    converted.append(None)
260                else:
261                    raise ValueError(f"Unsupported type for to_datetime: {type(val)}")
262        return Series(converted, name=self.name, index=self.index)

311    def cut(self, bins=4, labels=None, precision=3, tie_breaker='upper'):
312        """
313        Bin the Series into discrete intervals.
314        
315        Parameters:
316        bins (int): Number of bins to create. Defaults to 4.
317        labels (list): Optional labels for the bins. If None, default labels will be used.
318        precision (int): Number of decimal places for bin edges. Defaults to 3.
319        tie_breaker (str): How to handle ties ('upper', 'lower', 'random'). Defaults to 'upper'.
320        
321        Returns:
322        Series: A new Series with binned data.
323        """
324        if not self.data:
325            return Series([], name=self.name, index=self.index)
326        
327        clean_data = [v for v in self.data if v is not None]
328        min_val, max_val = min(clean_data), max(clean_data)
329
330        if isinstance(bins, int):
331            step = (max_val - min_val) / bins
332            bin_edges = [round(min_val + i * step, precision) for i in range(bins + 1)]
333        else:
334            bin_edges = bins
335
336        def assign_bin(val):
337            if tie_breaker == 'lower' and val == bin_edges[0]:
338                return labels[0] if labels else 0
339            
340            
341            for i in range(len(bin_edges) - 1):
342                left = bin_edges[i]
343                right = bin_edges[i + 1]
344                if tie_breaker == 'upper':
345                    if left <= val < right:
346                        return labels[i] if labels else i
347                else:
348                    if left < val <= right:
349                        return labels[i] if labels else i
350            if val == bin_edges[-1]:
351                return labels[-1] if labels else len(bin_edges) - 2
352            return None
353        
354        binned = [assign_bin(v) if v is not None else None for v in self.data]
355        return Series(binned, name=self.name, index=self.index)

357    def rank(self, method='average', ascending=True):
358        """
359        Compute numerical data ranks (1 through n) along the Series.
360        
361        Parameters:
362        - method: {'average', 'min', 'max', 'first', 'dense'}, default 'average'
363        - ascending: boolean, default True
364        
365        Returns:
366        Series: A new Series with ranked values.
367        """
368        values = self.data
369        indexed = list(enumerate(values))
370
371        if not ascending:
372            indexed.sort(key=lambda x: -x[1])
373        else:
374            indexed.sort(key=lambda x: x[1])
375
376        ranks = [0] * len(values)
377        cur_rank = 1
378        dense_rank = 1
379        i = 0
380
381        while i < len(indexed):
382            j = i
383            # find group of tied values
384            while j + 1 < len(indexed) and indexed[j + 1][1] == indexed[i][1]:
385                j+= 1
386
387            group = indexed[i:j + 1]
388            indices = [idx for idx, _ in group]
389            group_size = len(group)
390
391            if method == 'average':
392                avg_rank = sum(range(cur_rank, cur_rank + group_size)) / group_size
393                for idx in indices:
394                    ranks[idx] = avg_rank
395            elif method == 'min':
396                for idx in indices:
397                    ranks[idx] = cur_rank
398            elif method == 'max':
399                for idx in indices:
400                    ranks[idx] = cur_rank + group_size - 1
401            elif method == 'first':
402                for offset, (idx, _) in enumerate(group):
403                    ranks[idx] = cur_rank + offset
404            elif method == 'dense':
405                for idx in indices:
406                    ranks[idx] = dense_rank
407
408            i = j + 1
409            cur_rank += group_size
410            if method == 'dense':
411                dense_rank += 1
412
413        return Series(ranks, name=f"{self.name}_rank", index=self.index)

433    def quantile(self, q):
434        """
435        Compute the q-th quantile of the Series.
436        
437        Parameters:
438        q (float): The quantile to compute (0 <= q <= 1).
439        
440        Returns:
441        float: The q-th quantile value.
442        """
443        if not self.data:
444            return None if isinstance(q, float) else [None for _ in q]
445        
446        sorted_data = sorted(x for x in self.data if x is not None)
447        n = len(sorted_data)
448
449        def compute_single_quantile(p):
450            if not 0 <= p <= 1:
451                raise ValueError("Quantile must be between 0 and 1.")
452            idx = p * (n-1)
453            lower = int(idx)
454            upper = min(lower + 1, n-1)
455            weight = idx - lower
456            return sorted_data[lower] * (1 - weight) + sorted_data[upper] * weight
457        
458        if isinstance(q, list):
459            return [compute_single_quantile(p) for p in q]
460        return compute_single_quantile(q)

462    def percentile(self, p):
463        """ Equivalent to quantile"""
464        if isinstance(p, list):
465            return self.quantile([x/100 for x in p])
466        return self.quantile(p/100)

312    def cut(self, bins=4, labels=None, precision=3, tie_breaker='upper'):
313        """
314        Bin the Series into discrete intervals.
315        
316        Parameters:
317        bins (int): Number of bins to create. Defaults to 4.
318        labels (list): Optional labels for the bins. If None, default labels will be used.
319        precision (int): Number of decimal places for bin edges. Defaults to 3.
320        tie_breaker (str): How to handle ties ('upper', 'lower', 'random'). Defaults to 'upper'.
321        
322        Returns:
323        Series: A new Series with binned data.
324        """
325        if not self.data:
326            return Series([], name=self.name, index=self.index)
327        
328        clean_data = [v for v in self.data if v is not None]
329        min_val, max_val = min(clean_data), max(clean_data)
330
331        if isinstance(bins, int):
332            step = (max_val - min_val) / bins
333            bin_edges = [round(min_val + i * step, precision) for i in range(bins + 1)]
334        else:
335            bin_edges = bins
336
337        def assign_bin(val):
338            if tie_breaker == 'lower' and val == bin_edges[0]:
339                return labels[0] if labels else 0
340            
341            
342            for i in range(len(bin_edges) - 1):
343                left = bin_edges[i]
344                right = bin_edges[i + 1]
345                if tie_breaker == 'upper':
346                    if left <= val < right:
347                        return labels[i] if labels else i
348                else:
349                    if left < val <= right:
350                        return labels[i] if labels else i
351            if val == bin_edges[-1]:
352                return labels[-1] if labels else len(bin_edges) - 2
353            return None
354        
355        binned = [assign_bin(v) if v is not None else None for v in self.data]
356        return Series(binned, name=self.name, index=self.index)

358    def rank(self, method='average', ascending=True):
359        """
360        Compute numerical data ranks (1 through n) along the Series.
361        
362        Parameters:
363        - method: {'average', 'min', 'max', 'first', 'dense'}, default 'average'
364        - ascending: boolean, default True
365        
366        Returns:
367        Series: A new Series with ranked values.
368        """
369        values = self.data
370        indexed = list(enumerate(values))
371
372        if not ascending:
373            indexed.sort(key=lambda x: -x[1])
374        else:
375            indexed.sort(key=lambda x: x[1])
376
377        ranks = [0] * len(values)
378        cur_rank = 1
379        dense_rank = 1
380        i = 0
381
382        while i < len(indexed):
383            j = i
384            # find group of tied values
385            while j + 1 < len(indexed) and indexed[j + 1][1] == indexed[i][1]:
386                j+= 1
387
388            group = indexed[i:j + 1]
389            indices = [idx for idx, _ in group]
390            group_size = len(group)
391
392            if method == 'average':
393                avg_rank = sum(range(cur_rank, cur_rank + group_size)) / group_size
394                for idx in indices:
395                    ranks[idx] = avg_rank
396            elif method == 'min':
397                for idx in indices:
398                    ranks[idx] = cur_rank
399            elif method == 'max':
400                for idx in indices:
401                    ranks[idx] = cur_rank + group_size - 1
402            elif method == 'first':
403                for offset, (idx, _) in enumerate(group):
404                    ranks[idx] = cur_rank + offset
405            elif method == 'dense':
406                for idx in indices:
407                    ranks[idx] = dense_rank
408
409            i = j + 1
410            cur_rank += group_size
411            if method == 'dense':
412                dense_rank += 1
413
414        return Series(ranks, name=f"{self.name}_rank", index=self.index)

435        """
436        Compute the q-th quantile of the Series.
437        
438        Parameters:
439        q (float): The quantile to compute (0 <= q <= 1).
440        
441        Returns:
442        float: The q-th quantile value.
443        """
444        if not self.data:

446        
447        sorted_data = sorted(x for x in self.data if x is not None)
