 31        """Read a CSV file or string into a dataset.
 32        
 33        Parameters:
 34        -----------

 36            from_string: (bool): If True, treats path_or_str as a string, otherwise as a file path.
 37            encoding: (str): Encoding to use when reading the file
 38            converters: (dict): Optional dict of colum: function
 39            usecols: (list): Optionaal list of columns to keep
 40            
 41        Returns:
 42        -----------
 43            DataSet: A DataSet object containing the data from the CSV.
 44        """
 45        if from_string:
 46            f = io.StringIO(path_or_str)
 47        else:
 48            f = open(path_or_str, newline='')
 49
 50        reader = csv.DictReader(f)
 51        rows = []
 52
 53        # attempt numeric conversion
 54        for row in reader:
 55            parsed_row = {}
 56            for k, v in row.items():
 57                if usecols and k not in usecols:
 58                    continue
 59
 60                # Handle custom converter
 61                if converters and k in converters:
 62                    try:
 63                        parsed_row[k] = converters[k](v)
 64                        continue
 65                    except Exception:
 66                        parsed_row[k] = v
 67                        continue
 68
 69                # Handle datetime parsing
 70                if parse_dates and k in parse_dates:
 71                    try:
 72                        parsed_row[k] = datetime.fromisoformat(v)
 73                        continue
 74                    except ValueError:
 75                        try:
 76                            parsed_row[k] = datetime.strptime(v, "%Y-%m-%d")
 77                        except:
 78                            parsed_row[k] = v
 79                        continue
 80
 81                # Try numeric fallback
 82                try:
 83                    parsed_row[k] = float(v) if '.' in v else int(v)
 84                except:
 85                    parsed_row[k] = v
 86
 87            rows.append(parsed_row)
 88                
 89        return Dataset(rows)

105    def read_sql(query: str, conn: Union[Engine, sqlite3.Connection, psycopg2.extensions.connection], index_col=None) -> "DataSet":
106        """
107        Execute SQL and convert result to Atrax DataSet, without using pandas.
108
109        Parameters:
110        -----------
111        query : str
112            SQL query to run.
113        conn : SQLAlchemy engine/connection, sqlite3, or psycopg2 connection.
114        index_col : str, optional
115            Optional column to set as index.
116
117        Returns:
118        --------
119        DataSet
120        """
121        try:
122            # Handle SQLAlchemy connection
123            if isinstance(conn, (Engine)):
124                with conn.connect() as connection:
125                    result = connection.execute(text(query))
126                    columns = result.keys()
127                    rows = result.fetchall()
128
129            # Handle sqlite3 and psycopg2
130            else:
131                cursor = conn.cursor()
132                cursor.execute(query)
133                columns = [desc[0] for desc in cursor.description]
134                rows = cursor.fetchall()
135
136            # Convert rows into columns
137            data = {col: [] for col in columns}
138            for row in rows:
139                for col, val in zip(columns, row):
140                    data[col].append(val)
141
142            # Set index if requested
143            if index_col and index_col in data:
144                index = data.pop(index_col)
145                return Dataset(data, index=index_col, index_values=index)
146            else:
147                return Dataset(data)
148
149        except Exception as e:
150            raise RuntimeError(f"Failed to execute query: {e}")

150    def _infer_dtype(self):
151        if all(isinstance(x, float) for x in self.data):
152            return 'float64'
153        elif all(isinstance(x, bool) for x in self.data):
154            return 'bool'
155        elif all(isinstance(x, int) for x in self.data):
156            return 'int64'
157        elif all(isinstance(x, str) for x in self.data):
158            return 'str'

162            return 'object'

165        if isinstance(other, Series):

167                raise ValueError(f"Series must have the same length. found: {len(self.data)} and {len(other.data)}")
